          #include <Servo.h> 
          Servo servo3;

          void setup() { 
             Serial.begin(9600);
             servo3.attach(3); 
             servo5.attach(5); 
             servo6.attach(6); 
             servo9.attach(9); 
          } 
            
          void loop() { 
             int val0 = analogRead(A0); 
             int val1 = analogRead(A1); 
             int val2 = analogRead(A2); 
             int val3 = analogRead(A3); 
              
             Serial.print(val0); 
             Serial.print(","); 
             Serial.print(val1); 
             Serial.print(","); 
             Serial.print(val2); 
             Serial.print(","); 
             Serial.print(val3); 
             Serial.println(); 
            
             //750, val0 
             if(val0 > 650) { 
              servo3.write(10); 
             } else{ 
              servo3.write(100); 
             } 
            
             //740, val3 
             if(val3 > 740) { 
              servo5.write(5); 
             } else{ 
              servo3.write(90); 
             } 
            
             //930 - val 2 
             if(val2 > 930) { 
              servo6.write(10); 
             } else{ 
              servo6.write(110); 
             } 
            
             //val1 - 650 
             if(val1 > 650) { 
              servo9.write(10); 
             } else{ 
              servo9.write(100); 
             } 
            
             delay(15); 
          }
